#include <stdio.h>
#include <stdlib.h>
#include <iostream>

using namespace std;

// Data members
// Programa Ref_086_data_private3

class pessoa
{
	// private data por defeito

	unsigned short int idade;
	
	char *name; 

public:

	// serve para atribuir (Set) valor ao data member idade	
	void setIdade(unsigned short int i)
	{
		idade = i;
	}

	// serve para ler (Get) o valor do data member idade	
	int getIdade()
	{
		return idade;
	}

	// serve para atribuir (Set) valor ao data member nome
	void setNome(char *p)
	{
		name = p;
	}

	// serve para ler (Get) o valor do data member nome
	char * getNome()
	{
		return name;
	}
};

int main()
{
	pessoa pess; // objeto pess

	char nome[30];
	
// atribui valor ao private data member idade
// através de um método público
	
	pess.setIdade(25);

// print a idade
	cout << endl << endl << "Idade = " 
				 << pess.getIdade() << endl;

cout << "\nDigite o nome: ";
cin >> nome;
	
// print o nome
	pess.setNome(nome);
	cout << endl << "Nome = " 
				 << pess.getNome() 
				 << endl << endl;

	cout << endl << "Digite uma tecla para sair"	
				 << endl << endl;
	fflush(stdin);
	getchar();

	return 0;
}
