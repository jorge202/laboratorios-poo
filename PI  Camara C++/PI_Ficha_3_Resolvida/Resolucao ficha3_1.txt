#include <stdio.h>
#include <stdlib.h>


void inverte_e_imprime_str(char str[]); // Protótipo da função

int main()
{
	char msg[40];

	printf("\n\nDigite uma string:\n\n");
	fflush(stdin);
	gets(msg);

// chamar uma função para inverter e imprimir a string digitada

	inverte_e_imprime_str(msg);

	printf("\n\n\n");
	system("pause");

	return 0;

} // Fecha main


void inverte_e_imprime_str(char str[])
{
	int i; 

// percorrer a string até ao fim
	i = 0;
	while (str[i] != '\0')
		i++;

	i--; // coloca-se no carater anterior a '\0'

// neste momento a var i está posicionada no fim da string
	printf("\n\nEis a string invertida: \n\n");
	while (i >= 0)
	{
		printf("%c", str[i]);
		i--;
	}
}
