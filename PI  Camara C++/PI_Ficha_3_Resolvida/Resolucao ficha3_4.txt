  Escreva um programa que leia uma string digitada e que, seguidamente,
  crie uma outra string-cópia, mas sem espaços.

  Mostre no ecrã a nova string. 

  Aplique somente manipulação carácter a carácter.
*/

#include <stdio.h>
#include <stdlib.h>

#define NUM_MAX_CARACT 40

int main()
{
	char strBase[NUM_MAX_CARACT];
	char strCopia[NUM_MAX_CARACT];

	int i,k;
	
	printf("\n\nDigite a string base: ");
	fflush(stdin);
	gets(strBase);

	i = 0;
	k = 0;
// Repetir enquanto não for encontrado o fim da string base
	while (strBase[i] != '\0')
	{
		if (strBase[i] != ' ')
		{
			strCopia[k++] = strBase[i];
		}
		i++; // actualiza contador da strBase
	}
	
// agora torna-se necessário terminar adequadamente a 
// strCopia com '\0'

	strCopia[k] = '\0';
	
	printf("\n\nA String Base: %s", strBase);
	printf("\n\nA String Copia-sem Espacos: %s", strCopia);

	printf("\n\n\n");
	system("pause");

	return 0;
}
